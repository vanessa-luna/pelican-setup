                                          ---
                                          var $fotoramaDiv = $('.fotorama');
                                          $fotoramaDiv.on('fotorama:loadvideo', function(e, fotorama, extra){
                                          setTimeout(function(){
                                          var $ibody = $('iframe', '.fotorama__video').contents().find('body');
                                          var $video = $ibody.find('video');
                                          // set loop attribute
                                          $video.attr({
                                          loop: true
                                          });
                                          }, 1000) // delay 1000ms, but not shorter as 100ms is very important!
                                          });
